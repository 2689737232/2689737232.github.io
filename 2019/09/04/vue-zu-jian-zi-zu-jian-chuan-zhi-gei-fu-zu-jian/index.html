子组件向父组件通信、传递数据

通过$children
 
<div id="app">
    <yes></yes>
</div>
    const vm = new Vue({
        el: "#app",
        mounted() {
            //通过ref拿到子组件的实例，进一步拿到子组件里面的值
            let data = this.$children[0].data;                
        },
        components: {
            yes: {
                data() {
                    return {
                        data: "我是子组件的数据"
                    }
                },
                template: "<p >我是组件</p>"
            }
        }
    })
通过引用的方式
 
<div id="app">
    <!-- 如果ref重名了，后面的会覆盖前面的。 -->
    <!-- 如果for循环 <p ref="lala" v-for="...."></p>。$refs会用一个数组存放这个ref -->
    <p ref="lala"></p>
    <yes ref='yes'></yes>
</div>
    const vm = new Vue({
        el: "#app",
        mounted() {
            //通过ref拿到子组件的实例，进一步拿到子组件里面的值
            let data = this.$refs.yes.data;                
            console.log(data);
        },
        components: {
            yes: {
                data() {
                    return {
                        data: "我是子组件的数据"
                    }
                },
                template: "<p >我是组件</p>"
            }
        }
    })

通过子组件内部事件

 <div id="app">
    <!-- 绑定函数 -->
    <yes :func="func"></yes>
</div>
const vm = new Vue({
    el: "#app",
    methods: {
        func(data) {
            console.log(data);
        }
    },
    components: {
        yes: {
            //通过属性拿到父组件的方法
            props: ["func"],
            data() {
                return {
                    data: "我是子组件的数据"
                }
            },
            template: "<button @click='innerFunc'>点击</button>",
            methods: {
                // 通过子组件的方法触发父组件的方法
                innerFunc() {
                    // props里面的属性可以直接拿来用。挂载到这个组件上面的
                    this.func(this.data);
                }
            }
        }
    }
})

通过子组件事件 $listeners

 <div id="app">
    <!-- 在子组件身上绑定事件 因为非原生dom点击yes元素func不会触发-->
    <yes @click="func"></yes>
</div>
const vm = new Vue({
    el: "#app",
    methods: {
        func(data) {
            console.log(data);
        }
    },
    components: {
        yes: {
            data() {
                return {
                    data: "我是子组件的数据"
                }
            },
            template: "<button @click='innerFunc'>点击</button>",
            methods: {
                innerFunc() {
                    // 如果在子组件上面通过v-on绑定事件，会记录到实例上的 $listeners 属性上。
                    //相当于执行了父组价里面的click方法
                    this.$listeners.click(this.data);

                    //也可以使用 $emit触发
                    this.$emit('click',this.data);
                }
            }
        }
    }
})
